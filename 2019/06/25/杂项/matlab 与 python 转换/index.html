取最大值索引

% matlab
[~,y(i)]=max(x(i,:));    % 取最大值的索引

# python
y[i-1] = np.argmax(x[i-1])

python 与 matlab 产生相同的随机数

rand('twister', 1337);
b2=2*rand(N1+1,N2)-1;

np.random.rand(1337)
# numpy产生的随机数是按列排的，需要转置两次
b2 = 2 * np.random.rand(N2, N1 + 1) - 1
b2 = b2.T
